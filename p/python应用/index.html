一、requests库核心用法

requests是Python中处理HTTP请求的第三方库，相比内置的urllib更简洁、易用，是爬虫开发的基础工具。

HTTP请求主要有GET（获取数据）、POST（提交数据）两类，requests库对其做了极简封装。

# 基础GET请求

# 发送请求并获取响应

# print("二进制内容：", response.content)    # 图片/视频等二进制数据（如爬取图片时用）

import requests
import re  # 正则表达式库

# 步骤1：确定目标URL和请求头
url = "https://www.baidu.com"
headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36"
}

# 步骤2：发送请求
try:
    response = requests.get(url, headers=headers, timeout=5)
    response.raise_for_status()
    response.encoding = "utf-8"
    html = response.text  # 获取网页源代码

    # 步骤3：解析数据（正则提取<title>标签内容）
    title_pattern = re.compile(r"<title>(.*?)</title>")  # 匹配标题的正则
    title = title_pattern.search(html).group(1)  # 提取标题
    print("百度首页标题：", title)

    # 步骤4：保存数据（保存到txt文件）
    with open("baidu_title.txt", "w", encoding="utf-8") as f:
        f.write(f"标题：{title}\n爬取时间：{response.headers['Date']}")
    print("数据保存成功！")

except Exception as e:
    print("爬取失败：", e)

三、爬虫注意事项（必看）

  1. 遵守网站规则：查看目标网站的robots.txt（如https://www.baidu.com/robots.txt），遵守爬取限制；
  2. 控制爬取速度：添加延时（time.sleep(1)），避免频繁请求导致IP被封；
  3. 合法合规：不爬取敏感数据、付费内容，不用于商业用途；
  4. 反爬应对：合理使用代理IP、Cookie、请求头，避免单一IP高频访问。

import os
import shutil
from pathlib import Path

# 场景1：遍历指定目录下所有txt文件
def find_txt_files(folder_path):
    txt_files = []
    # 方式1：os.walk 遍历
    for root, dirs, files in os.walk(folder_path):
        for file in files:
            if file.endswith(".txt"):
                txt_files.append(os.path.join(root, file))
    # 方式2：pathlib（更简洁）
    # txt_files = [str(p) for p in Path(folder_path).rglob("*.txt")]
    return txt_files

# 场景2：复制/移动文件
def copy_files(source_file, target_folder):
    # 确保目标文件夹存在
    os.makedirs(target_folder, exist_ok=True)
    # 复制文件
    shutil.copy(source_file, target_folder)
    print(f"文件 {source_file} 已复制到 {target_folder}")

# 场景3：读取/写入文件
def file_operation(file_path):
    # 读取文件（按行读取）
    with open(file_path, "r", encoding="utf-8") as f:
        content = f.readlines()
        print("文件内容：", content)
    
    # 写入文件（追加模式）
    with open(file_path, "a", encoding="utf-8") as f:
        f.write("\n新增的脚本内容")

# 测试调用
if __name__ == "__main__":
    # 替换为你的本地路径
    test_folder = "./test_dir"
    test_file = "./test.txt"
    
    # 先创建测试文件
    with open(test_file, "w", encoding="utf-8") as f:
        f.write("初始内容")
    
    # 调用函数
    print("目录下的txt文件：", find_txt_files("./"))
    copy_files(test_file, test_folder)
    file_operation(test_file)

import json

# 场景：JSON字符串 ↔ Python字典互转
def json_operation():
    # 1. Python字典转JSON字符串
    data = {"name": "张三", "age": 25, "hobbies": ["编程", "阅读"]}
    json_str = json.dumps(data, ensure_ascii=False, indent=4)  # ensure_ascii=False显示中文
    with open("data.json", "w", encoding="utf-8") as f:
        f.write(json_str)
    
    # 2. 读取JSON文件转Python字典
    with open("data.json", "r", encoding="utf-8") as f:
        load_data = json.load(f)
    print("读取的JSON数据：", load_data["name"])

import subprocess

# 场景：执行cmd/终端命令（如查看目录、ping测试）
def run_system_cmd(cmd):
    try:
        # 执行命令并获取输出
        result = subprocess.run(
            cmd, 
            shell=True,  # Windows需开启
            capture_output=True,  # 捕获输出
            encoding="utf-8"
        )
        if result.returncode == 0:  # 0表示执行成功
            print("命令执行结果：", result.stdout)
        else:
            print("命令执行失败：", result.stderr)
    except Exception as e:
        print("异常：", e)

# 测试

import pyautogui
import time

# 场景：自动打开记事本并输入内容
def auto_desktop():
    # 设置操作间隔（避免速度过快）
    pyautogui.PAUSE = 1
    
    # 1. 打开运行窗口（Win+R）
    pyautogui.hotkey("win", "r")
    # 2. 输入notepad并回车
    pyautogui.typewrite("notepad")
    pyautogui.press("enter")
    # 3. 等待记事本打开
    time.sleep(2)
    # 4. 输入内容
    pyautogui.typewrite("Python脚本自动输入的内容！")

# 运行前确保桌面无遮挡

import requests

# 场景：批量下载图片
def download_images(img_urls, save_folder):
    os.makedirs(save_folder, exist_ok=True)
    for idx, url in enumerate(img_urls):
        try:
            response = requests.get(url, timeout=10)
            response.raise_for_status()
            # 保存图片
            with open(f"{save_folder}/img_{idx+1}.jpg", "wb") as f:
                f.write(response.content)
            print(f"图片{idx+1}下载完成")
        except Exception as e:
            print(f"图片{idx+1}下载失败：", e)

三、脚本开发必备技巧

1. 脚本调试技巧

  • 使用 print() 或 logging 库打印关键变量（推荐 logging，可输出到文件）；
  • 用 if __name__ == "__main__" 隔离测试代码，避免脚本被导入时执行；
  • 分步执行：先测试单个功能，再整合整体逻辑。

2. 脚本健壮性提升

  • 增加异常处理（try-except），避免脚本因小错误崩溃；
  • 对文件路径、网络请求等添加“存在性检查”（如 os.path.exists()）；
  • 关键操作添加日志记录，方便排查问题。
